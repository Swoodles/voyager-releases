🌐 Connect Online
Voyager uses Supabase (free) for friends and trip sharing. You need to create a free project at supabase.com and run the setup SQL once. See instructions below.
📋 Setup SQL — run once in Supabase SQL Editor 
create table if not exists voyager_users (
  id uuid default gen_random_uuid() primary key,
  username text unique not null,
  display_name text, avatar_emoji text default '✈️',
  bio text, created_at timestamptz default now()
);
create table if not exists voyager_friends (
  id uuid default gen_random_uuid() primary key,
  user_id uuid references voyager_users(id) on delete cascade,
  friend_id uuid references voyager_users(id) on delete cascade,
  status text default 'pending',
  created_at timestamptz default now(),
  unique(user_id, friend_id)
);
create table if not exists voyager_shared_trips (
  id uuid default gen_random_uuid() primary key,
  from_user_id uuid references voyager_users(id) on delete cascade,
  to_user_id uuid references voyager_users(id) on delete cascade,
  trip_name text, trip_data jsonb, message text,
  created_at timestamptz default now()
);
alter table voyager_users enable row level security;
alter table voyager_friends enable row level security;
alter table voyager_shared_trips enable row level security;
create policy "public read users" on voyager_users for select using (true);
create policy "public insert users" on voyager_users for insert with check (true);
create policy "public read friends" on voyager_friends for select using (true);
create policy "manage own friends" on voyager_friends for all using (true);
create policy "public read shared trips" on voyager_shared_trips for select using (true);
create policy "insert shared trips" on voyager_shared_trips for insert with check (true);
